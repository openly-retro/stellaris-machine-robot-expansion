#Reanimated Armies: Reanimating the enemy's army as your own
#An army has been killed in ground combat
# This = owner
# From = army
# FromFrom = opponent
# FromFromFrom = planet
country_event = {
	id = oxr_mdlc.900
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = fromfrom
		fromfrom = {
			# has_valid_civic = civic_reanimated_armies
			has_valid_civic = xvcv_mdlc_civic_assimilating_swarm
			has_ascension_perk = ap_mechromancy
		}
		from = {
			exists = species
			species = {
				is_robotic = no
			}
		}
		NOT = { is_country_type = "gate_builders" }
	}

	immediate = {
		random_list = {
			2 = { }
			1 = {
				fromfromfrom = {
					create_army = {
						owner = root.fromfrom
						# type = oxr_mdlc_reanimated_cyborg_army
						type = undead_army
						species = root.from.species
					}
				}
			}
		}
	}
}
