#set country flags for mod configurations (Machine or robot empire only)

	id = xvcv_mdlc_config_effect.1
	hide_window = yes
	is_triggered_only = yes

		set_country_flag = xvcv_mdlc_civic_assimilating_swarm_message_on

		set_country_flag = xvcv_mdlc_decision_machine_cpu_project_message_on
		set_country_flag = xvcv_mdlc_decision_machine_ser_project_message_on
		set_country_flag = xvcv_mdlc_abyss_access_possible_message_on
		set_country_flag = xvcv_mdlc_decision_virtual_world_project_message_on
	}
}
#from robot to machine species conversion (Gestalt empires only)
country_event = {
	id = xvcv_mdlc_config_effect.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ai = no
		has_country_flag = xvcv_mdlc_config_1_on
		has_authority = auth_machine_intelligence
	}

	immediate = {
		every_owned_species = {
			limit = { has_trait = trait_mechanical }
			if = {
				limit = { root = { has_country_flag = xvcv_mdlc_config_3_on } }
				root = {
					every_owned_pop_group = {
						limit = { is_same_species = prevprev }
						change_species = prev.species
					}
				}
				else = {
					create_species = {
						adjective = this
						name = this
						namelist = root
						plural = this
						class = "MACHINE"
						portrait = this
						traits = { trait = trait_machine_unit }
						traits = this
						homeworld = root.capital_scope
						effect = {
							change_species_characteristics = {
								remove_trait = trait_mechanical
							}
							# oxr_mdlc_set_custom_species_habitability = yes
						}
					}
					root = {
						every_owned_pop_group = {
							limit = { is_same_species = prevprev }
							change_species = last_created_species
						}
					}
					xvcv_mdlc_machine_species_rights_adjustments_effect = yes
				}
			}
		}
		log = "xvcv_mdlc_config_effect.2: from robot to machine species conversion was called"
	}
}

#species assimilation
country_event = {
	id = xvcv_mdlc_config_effect.4
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ai = no
		has_country_flag = xvcv_mdlc_config_3_on
		is_robot_empire = yes
		any_owned_species = {
			NOT = { is_same_species = prev.species }
			OR = {
				has_trait = trait_machine_unit
				has_trait = trait_mechanical
			}
		}
	}

	immediate = {
		every_owned_species = {
			limit = {
				NOT = { is_same_species = prev.species }
				OR = {
					has_trait = trait_machine_unit
					has_trait = trait_mechanical
				}
			}
			root = {
				random_owned_pop_group = {
					limit = { is_same_species = prevprev }
					modify_species = {
						species = this
						base = prev.species
					}
				}
				every_owned_pop_group = {
					limit = { is_same_species = prevprev }
					change_species = last_created_species
				}
			}
		}
		log = "xvcv_mdlc_config_effect.4 (species assimilation) was called"
	}
}

#auto pop migration
country_event = {
	id = xvcv_mdlc_config_effect.5
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ai = no
		has_country_flag = xvcv_mdlc_config_4_on
		count_owned_colony = { count > 1 }
		is_robot_empire = yes
		any_owned_pop_group = {is_unemployed = yes }
		any_owned_colony = {
			free_jobs > 100
			is_under_colonization = no
			has_orbital_bombardment = no
			is_controlled_by = owner #owner must be controller
		}
	}

	# Retro: Move batches of 100, and ignore leftovers (sorry leftovers..)
	immediate = {

		while = {
			limit = {
				any_owned_pop_group = {
					is_unemployed = yes
					pop_group_size > 100
				}
				any_owned_colony = {
					free_jobs > 100
					is_under_colonization = no
					has_orbital_bombardment = no
					is_controlled_by = owner
				}
			}
			random_owned_pop_group = {
				limit = {
					is_unemployed = yes
					pop_group_size > 100
				}

				prev = {
					# Country scope
					random_owned_colony = {
						limit = {
							free_jobs > 0
							is_under_colonization = no
							has_orbital_bombardment = no
							is_controlled_by = owner
						}
						save_event_target_as = xvcv_mdlc_colony_unemployment_resettlement_target
					}
				}
				# Now we are back in pop group scope
				# resettle_pop = { pop = prevprev planet = this }
				# It doesn't really resettle pop groups ... it's more like "cut and paste"
				oxr_resettle_pop_group = {
					source_pop_group = this
					target_planet = event_target:xvcv_mdlc_colony_unemployment_resettlement_target
				}
				log = "xvcv_mdlc_config_effect.5: Pop has been resettled to \\[This.GetName]"
			}
		}
	}
}
