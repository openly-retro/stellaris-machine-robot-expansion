#This is for the four SR buildings since stellaris as of 4.4 doesn't support hidden effects in building's triggered_planet_modifier
rikk_mdlc_planet_effect_add_SR_deposit = {
	planet = {
		while = {
			limit = { has_deposit = rikk_mdlc_d_world_machine_special_resources }
			every_deposit = {
				limit = { is_deposit_type = rikk_mdlc_d_world_machine_special_resources }
				remove_deposit = yes
			}
		}
		if = {
			limit = { 
				NOT = { has_deposit = rikk_mdlc_d_world_machine_special_resources }
				colony = {
					OR = {
						has_building = oxr_mdlc_world_machines_building_welding_colony
						has_building = oxr_mdlc_world_machines_building_particle_collider
						has_building = oxr_mdlc_world_machines_building_crystal_cultivation_tank
						has_building = rikk_mdlc_world_machines_building_ancient_refinery
					}
				}
			}
			add_deposit = rikk_mdlc_d_world_machine_special_resources
		}
	}
}
