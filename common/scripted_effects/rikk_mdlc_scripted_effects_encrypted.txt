#calculates the bonus from overclocked pops
rikk_mdlc_encrypted_inc_scrap = {
	if = {
		limit = { NOT = { is_variable_set = rikk_mdlc_encrypted_scrap_counter } }
		set_variable = {
			which = rikk_mdlc_encrypted_scrap_counter
			value = 0
		}
	}
	
	change_variable = {
		which = rikk_mdlc_encrypted_scrap_counter
		value = 1
	}
}
