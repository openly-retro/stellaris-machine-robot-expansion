remove_if_has_building = {
	optimize_memory
	if = {
		limit = { has_building = $WHICH$ }
		remove_building = $WHICH$
	}
}

remove_and_replace_building_if = {
	if = {
		limit = {
			has_building = $REPLACE$
		}
		remove_building = $REPLACE$
		add_building = $WITH$
	}
}


clear_variable_if_exists = {
	if = {
		limit = {
			is_variable_set = $WHICH$
		}
		clear_variable = $WHICH$
	}
}

set_or_increment_variable = {
	# Set variable to a value, if it doesn't exist, or increment by that amount
	if = {
		limit = {
			NOT = { is_variable_set = $VAR$ }
		}
		set_variable = {
			which = $VAR$
			value = $AMOUNT$
		}
	}
	else = {
		change_variable = {
			which = $VAR$
			value = $AMOUNT$
		}
	}
}

dec_var_by_one = {
	# Decrement variable by 1
	change_variable = {
		which = $which$
		value = -1
	}
}
add_one_to_var = {
	change_variable = {
		which = $which$
		value = 1
	}
}

set_scope_flag = {

	}

if_planet_doesnt_have_flag_then = {
	if = {
		limit = {
			planet = {
				NOT = {
					has_planet_flag = $FLAG$
				}
			}
		}
		$DO$ = yes
	}
}

# Default unalive 100 from the group
oxr_kill_pop_from_group = {
	kill_pop_group = {
		pop_group = this
		# amount = 100
		[[count] amount = $count$ ]
		[[!count] amount = 100 ]
	}
}

oxr_kill_all_pops_in_group = {
	kill_pop_group = {
		pop_group = this
		percentage = 1
	}
