# Pop group stuff
oxr_mdlc_pop_amount_change_category = {
	planet = {
		create_pop_group = {
			pop_group = root
			category = $CATEGORY$
			size = 0
			effect = {
				transfer_pop_amount = {
					source = root
					target = this
					amount = $AMOUNT$
				}
			}
		}
	}
}

oxr_mdlc_country_synth_recalculate_pop_growth_factor = {
	# Remove the country modifier,
	# export pop growth modifier to a var
	# reduce pop growth by that amount
	# increase pop assembly by that amount,
	# then add back the modifier

	remove_modifier = $MODIFIER$

	# Let's see if the game state can respect the change in the modifier in one day
	export_modifier_to_variable = {
		modifier = logistic_growth_mult
		variable = oxr_mdlc_country_logistic_growth_mult_var
	}
	export_modifier_to_variable = {
		modifier = planet_pop_assembly_mult
		variable = oxr_mdlc_country_planet_pop_assembly_mult
	}

	add_modifier = {
		modifier = $MODIFIER$
		days = -1
	}
}

oxr_mdlc_country_set_oxr_mdlc_synth_pop_assembly_support_control_default = {
	set_policy = {
		policy = oxr_mdlc_synth_pop_assembly_support_control
		option = oxr_mdlc_synth_pop_assembly_support_control_default
		cooldown = no
	}
	oxr_mdlc_country_synth_recalculate_pop_growth_factor = {
		MODIFIER = oxr_mdlc_synth_pop_assembly_support_reroute_growth_unchanged
	}
}
