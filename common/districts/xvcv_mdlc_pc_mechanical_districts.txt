	}

		potential = {
			exists = owner
			owner = { has_active_tradition = tr_prosperity_interstellar_franchising }
		}
		modifier = { job_clerk_add = 100 }
	}

	triggered_planet_modifier = {

			exists = owner

		}
		modifier = { planet_housing_add = 100 }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_technology = tech_housing_1 }
		}
		modifier = { planet_housing_add = 100 }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				NOT = { has_valid_civic = civic_agrarian_idyll }
				has_auth_machine_intelligence = no
			}
		}
		modifier = { planet_housing_add = 100 }
	}

		triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				has_auth_machine_intelligence = yes
			}
		}
		modifier = {
			planet_housing_add = 100
			job_maintenance_drone_add = 100
		}
	}

	## Descriptions
	# Triggered Machine descriptions

	}
