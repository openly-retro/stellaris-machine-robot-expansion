xvcv_mdlc_pc_virtual_district_primary = {

	base_buildtime = @xvcv_mdlc_district_time_4
	default_starting_district = yes

	show_on_uncolonized = {
		exists = from
		from = { has_ascension_perk = xvcv_mdlc_ap_virtual_worlds }
		is_xvcv_mdlc_pc_virtual = yes
	}
	potential = {
		exists = owner
		owner = { has_ascension_perk = xvcv_mdlc_ap_virtual_worlds }
		is_xvcv_mdlc_pc_virtual = yes
	}

	zone_slots = {
		slot_city_government
		slot_city_01
		slot_city_02
	}

	allow = {
		has_building = xvcv_mdlc_building_matter_data_converter
	}
	resources = {
		category = planet_districts
		cost = {
			minerals = @xvcv_mdlc_district_cost_5
			rare_crystals = 100
		}
		upkeep = {
			energy = 10
			rare_crystals = 1
		}
	}

	planet_modifier = {
		planet_housing_add = @oxr_mdlc_pc_virtual_district_planet_housing_amount
		job_xvcv_mdlc_virtual_merchant_add = @oxr_mdlc_pc_virtual_resource_jobs_amount
	}
	## Housing
		triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_agrarian_idyll }
		}
		modifier = { planet_housing_add = -100 }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_prosperity_public_works }
		}
		modifier = { planet_housing_add = 100 }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_prosperity_optimized_nexus }
		}
		modifier = { planet_housing_add = 100 }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_technology = tech_housing_1 }
		}
		modifier = { planet_housing_add = 100 }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_technology = tech_housing_2 }
		}
		modifier = { planet_housing_add = 100 }
	}
	# Desc

	triggered_desc = {
		text = job_xvcv_mdlc_virtual_merchant_effect_desc
	}

	######

	conversion_ratio = 1
	inline_script = planet/districts/convert_to/oxr_wm_district_primary

	triggered_name = {
		text = xvcv_mdlc_pc_virtual_district_trade
	}
	triggered_desc = {
		text = xvcv_mdlc_pc_virtual_district_trade_desc
	}
}
