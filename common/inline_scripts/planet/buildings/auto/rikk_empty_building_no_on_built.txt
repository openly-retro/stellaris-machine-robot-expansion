category = $CATEGORY$
potential = {
	exists = owner
	owner = { oxr_mdlc_planet_owner_can_build_wm_districts = yes }
	is_planet_world_machine = yes
	$POTENTIAL$
}
show_tech_unlock_if = { oxr_mdlc_planet_owner_can_build_wm_districts = yes }
destroy_trigger = { owner = { oxr_mdlc_planet_owner_can_build_wm_districts = no } }

base_buildtime = @b$TIER$_time
resources = {
	category = $ECONOMIC_CATEGORY$
	cost = {
		minerals = @b$TIER$_minerals
		$RARE_MATERIAL$ = $RARE_COST$
	}
	upkeep = {
		energy = $TIER$
		$RARE_MATERIAL$ = $RARE_UPKEEP$
	}
	$RESOURCES$
	inline_script = {
		script = planet/buildings/auto/jobs/rikk_mdlc_$JOB$
		JOBS = $JOBS$
	}
}
inline_script = {
	script = planet/buildings/auto/modifiers/rikk_mdlc_$MODIFIER$
	JOBS = $JOBS$
}
building_sets = {
	oxr_mdlc_building_set_wm_all
	oxr_mdlc_building_set_wm_$CATEGORY$
	oxr_mdlc_building_set_wm_$SET$
}
