# Swap out a generic deposit, which adds +1 max to all districts of a type,
# with a particular district once built
on_queued = {
	remove_deposit = $all$
	add_deposit = $which$
	$on_queued$
}
on_unqueued = {
	remove_deposit = $which$
	add_deposit = $all$
	$on_unqueued$
}
on_destroy = {
	remove_deposit = $which$
	add_deposit = $all$
	$on_destroy$
}
