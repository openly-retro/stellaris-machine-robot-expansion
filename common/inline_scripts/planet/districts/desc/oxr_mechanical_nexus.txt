triggered_name = {
	trigger = {
		is_homeworld = yes
		is_capital = yes
		exists = owner
		owner = { has_auth_machine_intelligence = yes }
	}
	text = city_machine_core_1
}

	trigger = {
		OR = {

					value > 0
				}

					type = zone_factory
					value > 0
				}
			}
		}
		exists = owner
		owner = { has_auth_machine_intelligence = yes }
