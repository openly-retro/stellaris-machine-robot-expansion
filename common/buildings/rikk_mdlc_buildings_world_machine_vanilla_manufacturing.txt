# Turn nanites into alloys
oxr_mdlc_world_machines_building_nanotech_cauldron = {
	base_buildtime = @b2_time

	category = manufacturing

	potential = {
		exists = owner
		owner = {
			has_tradition = tr_nanotech_2
			has_oxr_mdlc_origin_world_machine_awakened = yes
		}
		oxr_mdlc_planet_has_vanilla_pop_colony_modifier = no
		is_planet_world_machine = yes
	}

	allow = {
		is_planet_world_machine = yes
	}

	destroy_trigger = {
		exists = owner
		OR = {
			NOT = {
				owner = {
					has_tradition = tr_nanotech_2
					has_oxr_mdlc_origin_world_machine_awakened = yes
				}
			}
			oxr_mdlc_planet_has_vanilla_pop_colony_modifier = yes
		}
	}

	resources = {
		category = oxr_mdlc_world_machines_foundry_buildings
		cost = {
			alloys = @b3_minerals
		}
		upkeep = {
			nanites = 3
			energy = 30
		}
		produces = {
			alloys = 15
		}
	}

	building_sets = {
		oxr_mdlc_building_set_wm_all
		oxr_mdlc_building_set_wm_manufacturing
		oxr_mdlc_building_set_wm_nanotech
	}
}
