# Common categories for World Machines
xvcv_mdlc_world_machines_districts = {
	parent = planet_districts
	use_for_ai_budget = yes
	modifier_category = colony
	generate_mult_modifiers = {
		produces
		upkeep
		cost
	}
	# all add modifiers for this category are useless but good for hidden_effects where you just want it to look like all districts. I use it in the submods.
	generate_add_modifiers = {
		cost	
		produces 
		upkeep
	}
}
