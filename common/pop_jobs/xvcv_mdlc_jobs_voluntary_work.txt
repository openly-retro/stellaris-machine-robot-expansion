xvcv_mdlc_organic_foundry = {
	category = bio_trophy

	swappable_data = {
		default = {
			condition_string = SPECIALIST_JOB_TRIGGER
			building_icon = voluntary_work/xvcv_mdlc_building_organic_foundry
		}
	}

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
	}

	possible = {
		has_citizenship_type = { type = citizenship_organic_trophy }
	}

	tags = {
		xvcv_mdlc_organic_foundry	# job id for use with 'num_pops_with_job_tag'
		oxr_mdlc_job_tag_bio_trophy			# custom mod category
		job_tag_alloys				# I would like to standardize a job_tag prefix
		alloys
	}

	resources = {
		category = planet_bio_trophies
		produces = {
			trigger = {
				exists = owner
				owner = { is_catalytic_empire = no }
			}
			alloys = 3
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { is_catalytic_empire = no }
			}
			minerals = 6
		}
		produces = {
			trigger = {
				exists = owner
				owner = { is_catalytic_empire = yes }
			}
			alloys = 3.75
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { is_catalytic_empire = yes }
			}
			food = 9
		}
	}

	weight = {
		weight = @complex_drone_job_weight
		mult = value:job_weights_modifier|JOB|xvcv_mdlc_organic_foundry|RESOURCE|alloys|
		modifier = {
			factor = 5
			planet = {
				total_workforce_with_job_tag = {
					tags = { xvcv_mdlc_organic_foundry }
					value > 0
				}
			}
			# num_assigned_jobs = {
			# 	job = xvcv_mdlc_organic_foundry
			# 	value > 0
			# }
		}
	}
}
