# Jobs for virtual world

xvcv_mdlc_virtual_researcher = {
	category = specialist

	swappable_data = {
		default = {
			condition_string = SPECIALIST_JOB_TRIGGER
			building_icon = building_research_lab_1
		}
	}

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_specialist_job
	possible = {
		complex_specialist_job_check_trigger = yes
	}
	tags = {
		xvcv_mdlc_virtual_researcher
		oxr_mdlc_job_tag_virtual
		research
	}

	resources = {
		category = planet_researchers
		produces = {
			physics_research = 4
			engineering_research = 4
			society_research = 4
		}
		produces = {
			trigger = {
				exists = owner
				owner = { has_valid_civic = civic_technocracy }
			}
			unity = 1
		}
		upkeep = {
			consumer_goods = 2
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { has_valid_civic = civic_technocracy }
			}
			consumer_goods = 1
		}
	}

	weight = {
		weight = @complex_drone_job_weight
		mult = value:job_weights_research_modifier|JOB|xvcv_mdlc_virtual_researcher|
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 2
			has_living_standard = { type = living_standard_academic_privilege }
		}
		modifier = {
			factor = 0.1
			can_take_servant_job = yes
		}
		modifier = {
			factor = 1.5
			planet = {
				num_pops_with_job_tag = {
					tags = { xvcv_mdlc_virtual_researcher }
					value > 0
				}
			}
		}
		modifier = {
			factor = 0.8
			is_enslaved = yes
			has_slavery_type = { type = slavery_indentured }
		}
	}
}
xvcv_mdlc_virtual_merchant = {
	category = specialist

	swappable_data = {
		default = {
			condition_string = SPECIALIST_JOB_TRIGGER
			building_icon = building_galactic_stock_exchange
		}
	}

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_ruler_job

	tags = {
		xvcv_mdlc_virtual_merchant
		oxr_mdlc_job_tag_virtual
		trade
		amenities
	}

	resources = {
		category = planet_merchants
		produces = {
			trigger = {
				exists = owner
				owner = { has_civic = civic_merchant_guilds	}
			}
			unity = 2
		}
		produces = {
			trigger = {
				exists = owner
			}
			trade = 12
			mult = value:scripted_modifier_mult|MODIFIER|planet_jobs_trade_produces_mult|
		}
	}

	weight = {
		weight = @ruler_job_weight
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|planet_jobs_trade_produces_mult|
		modifier = {
			factor = 5
			oxr_mdlc_planet_any_pops_have_job_tag = { JOB = xvcv_mdlc_virtual_merchant }
		}
		# crisis purge
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				OR = {
					is_country_type = swarm
					is_country_type = ai_empire
				}
			}
			OR = { # contingency won't purge itself
				NOT = { exists = event_target:custodian_bot }
				AND = {
					exists = event_target:custodian_bot
					NOT = { species = { is_same_value = event_target:custodian_bot } }
				}
			}
		}
	}
}

				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
		}
		modifier = {
			factor = 1.5
			oxr_mdlc_planet_any_pops_have_job_tag = { JOB = xvcv_mdlc_virtual_entertainer }
		}
		modifier = {
			factor = 0.1
			is_domestic_servant = yes
		}
		modifier = {
			factor = 0.1
			NOT = {
				oxr_mdlc_planet_any_pops_have_job_tag = { JOB = xvcv_mdlc_virtual_entertainer }
			}
			exists = planet
			planet = {
				free_amenities > 5
			}
		}
		modifier = {
			factor = 2
			NOT = {
				oxr_mdlc_planet_any_pops_have_job_tag = { JOB = xvcv_mdlc_virtual_entertainer }
			}
			exists = planet
