rikk_mdlc_nano_drone = {
	category = maintenance_drone
	swappable_data = {
		default = {
			condition_string = DRONE_JOB_TRIGGER
			building_icon = building_maintenance_depot
		}
	}
	is_capped_by_modifier = no

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}
	tags = {
		rikk_mdlc_nano_drone
		oxr_mdlc_job_tag_pop_assembly
		pop_assembly
	}
	
	possible_precalc = can_fill_drone_job

	possible = {
        has_trait = rikk_mdlc_trait_robotic_constant_growth
    }

	planet_modifier = {
		planet_pop_assembly_mult = 0.001
		planet_jobs_produces_mult = 0.002
		planet_jobs_upkeep_mult = 0.001
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_amenities_no_happiness_add = 50
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

    # copy from vanilla maintenance drone
    # slightly lower weight than maintenance drone
    weight = {
		weight = 2  # @maintenance_drone_job
		# mult = value:scripted_modifier_job_weight_mult|MODIFIER|planet_pop_assembly_mult|

		# copy mult from vanilla maint drone job
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			mult = value:job_weights_modifier|JOB|maintenance_drone|RESOURCE|unity|FACTOR|0.25|
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_machine_maintenance_protocols
					has_active_tradition = tr_versatility_operational_proxies
				}
			}
		}
		# Respond to finer controls via planet decision
		modifier = {
			factor = 5
			exists = owner
			planet = {
				has_planet_flag = rikk_mdlc_planet_extended_nano_drone_subsidy
			}
		}
		modifier = {
			factor = 0
			exists = owner
			planet = {
				has_planet_flag = rikk_mdlc_planet_extended_nano_drone_deactivation
			}
		}
		# Respond to finer controls via policy changes
		# Respect planet preferences over empire preferences
		modifier = {
			factor = 5
			exists = owner
			owner = {
				has_policy_flag = oxr_mdlc_extended_traits_nano_drone_control_subsidize
			}
			planet = {
				NOT = {
					has_planet_flag = rikk_mdlc_planet_extended_nano_drone_subsidy
					has_planet_flag = rikk_mdlc_planet_extended_nano_drone_deactivation
				}
			}
		}
		modifier = {
			factor = 0
			exists = owner
			owner = {
				has_policy_flag = oxr_mdlc_extended_traits_nano_drone_control_decommission
			}
			planet = {
				# Do not override this planet's preference for nano drone jobs
				NOT = {
					has_planet_flag = rikk_mdlc_planet_extended_nano_drone_subsidy
				}
			}
		}
	}

}
