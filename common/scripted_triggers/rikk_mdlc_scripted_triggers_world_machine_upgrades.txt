rikk_mdlc_is_wm_planet_trigger = {
	optimize_memory
	OR = {
		is_xvcv_mdlc_pc_machine_auto = yes
		is_xvcv_mdlc_pc_machine_ecu = yes
		is_xvcv_mdlc_pc_machine_ser = yes
		is_xvcv_mdlc_pc_machine_cpu = yes
	}
}
rikk_mdlc_is_wm_industrial_trigger = {
	optimize_memory
	OR = {
		is_xvcv_mdlc_pc_machine_ecu = yes
		is_xvcv_mdlc_pc_machine_auto = yes
	}
}
rikk_mdlc_is_wm_computer_trigger = {
	optimize_memory
	OR = {
		is_xvcv_mdlc_pc_machine_ser = yes
		is_xvcv_mdlc_pc_machine_cpu = yes
	}
}
