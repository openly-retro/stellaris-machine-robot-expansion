		category = decisions

			energy = 4000
			rare_crystals = 800
			alloys = 400
			mult = value:scripted_modifier_mult|MODIFIER|rikk_mdlc_terraform_decisions_cost_mult|
		}
		cost = {
			trigger = { space_owner = { NOT = { has_tradition = rikk_mdlc_tr_geoscaping_2 } } }

		}

	potential = {
