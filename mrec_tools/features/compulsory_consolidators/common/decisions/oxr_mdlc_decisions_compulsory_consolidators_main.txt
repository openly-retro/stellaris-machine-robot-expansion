oxr_mdlc_decision_trigger_orbital_extraction_project = {

	enactment_time = 1
	resources = {
		category = decisions
		cost = {
			energy = 500
		}
	}
	potential = {
		exists = space_owner
		space_owner = { has_oxr_mdlc_civic_consolidators = yes }
	}
	allow = {
		# always = yes
		is_terraforming = no
		has_anomaly = no

		NOT = {
			has_planet_flag = oxr_mdlc_civic_consolidators_extraction_in_progress
			# Can't extract off the target planet
			oxr_mdlc_planet_is_empire_orbital_consolidator_target = no
			oxr_mdlc_planet_is_empire_planet_consolidation_target = no
		}
		oxr_mdlc_consolidators_planet_has_orbital_deposits = yes
	}

	effect = {
		log = "effect of oxr_mdlc_decision_trigger_orbital_extraction_project was fired"
		space_owner = {
			country_event = {
				id = oxr_mdlc_civic_consolidators.3000
				scopes = {
					from = prev
				}
			}
		}
	}
}
